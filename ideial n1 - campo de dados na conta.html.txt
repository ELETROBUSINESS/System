    <script>
        let currentRevenue = 4625.91;
        let isFaturamentoActive = true;
        let holdTimer = null;
        let holdDuration = 1200;
        let currentDueIndex = 0;
        const MAX_DOTS = 3;

        // Dados simulados (Ano Atual vs Ano Anterior)
        const currentYearData = [500, 1100, 850, 1600, 1300, 2400, 2100];
        const lastYearData = [450, 900, 700, 1500, 1400, 2100, 1800];

        function setViewMode(mode) {
            document.documentElement.setAttribute('data-view', mode);
            const chart = document.getElementById('chart-section');
            const history = document.getElementById('history-section');
            
            if(mode === 'desktop') {
                chart.style.display = 'block';
                history.style.display = 'block';
                fetchAreaChartData();
            } else {
                chart.style.display = 'none';
                history.style.display = 'none';
            }
            
            setTimeout(() => {
                const activeItem = document.querySelector('.nav-item.active');
                if (activeItem) {
                    const items = Array.from(document.querySelectorAll('.nav-item'));
                    navTo(activeItem, activeItem.querySelector('.nav-label').innerText, items.indexOf(activeItem));
                }
            }, 550);
        }

        function toggleSidebar() {
            const isMin = document.documentElement.getAttribute('data-minimized') === 'true';
            document.documentElement.setAttribute('data-minimized', !isMin);
            document.getElementById('toggle-icon').style.transform = isMin ? 'rotate(0deg)' : 'rotate(180deg)';
            setTimeout(() => {
                const activeItem = document.querySelector('.nav-item.active');
                if (activeItem) {
                    const items = Array.from(document.querySelectorAll('.nav-item'));
                    navTo(activeItem, activeItem.querySelector('.nav-label').innerText, items.indexOf(activeItem));
                }
            }, 400);
        }

        async function fetchAreaChartData() {
            const total = currentYearData.reduce((a, b) => a + b, 0);
            animateValue(0, total, 1500, (v) => {
                document.getElementById('chart-total-val').innerText = `R$ ${v.toLocaleString('pt-BR', {minimumFractionDigits: 2})}`;
            });
            renderAreaChart(currentYearData);
        }

        function renderAreaChart(values) {
            const linePath = document.getElementById('chart-line');
            const fillPath = document.getElementById('chart-fill');
            const dotsGroup = document.getElementById('chart-dots-group');
            const maxVal = Math.max(...values, ...lastYearData);
            const stepX = 400 / (values.length - 1);
            
            dotsGroup.innerHTML = '';
            
            let d = `M 0 ${160 - (values[0] / maxVal * 130)}`;
            createDot(0, 160 - (values[0] / maxVal * 130), 0);

            for(let i = 1; i < values.length; i++) {
                const x = i * stepX;
                const y = 160 - (values[i] / maxVal * 130);
                d += ` L ${x} ${y}`;
                createDot(x, y, i);
            }
            
            linePath.setAttribute('d', d);
            let fillD = d + ` L 400 160 L 0 160 Z`;
            fillPath.setAttribute('d', fillD);

            linePath.classList.remove('active');
            fillPath.classList.remove('active');
            void linePath.offsetWidth;
            linePath.classList.add('active');
            fillPath.classList.add('active');
        }

        function createDot(x, y, index) {
            const dotsGroup = document.getElementById('chart-dots-group');
            const circle = document.createElementNS("http://www.w3.org/2000/svg", "circle");
            circle.setAttribute("cx", x);
            circle.setAttribute("cy", y);
            circle.setAttribute("r", "4");
            circle.setAttribute("class", "chart-dot");
            
            circle.addEventListener('mouseenter', (e) => showTooltip(e, index));
            circle.addEventListener('mouseleave', hideTooltip);
            
            dotsGroup.appendChild(circle);
        }

        function showTooltip(e, index) {
            const tooltip = document.getElementById('chart-tooltip');
            const val = currentYearData[index];
            const oldVal = lastYearData[index];
            const diff = ((val - oldVal) / oldVal * 100).toFixed(1);
            const colorClass = diff >= 0 ? 'text-green-400' : 'text-red-400';
            const sign = diff >= 0 ? '+' : '';

            tooltip.innerHTML = `
                <div class="mb-1">R$ ${val.toLocaleString('pt-BR')}</div>
                <div class="${colorClass} font-bold">${sign}${diff}% vs ano ant.</div>
            `;
            
            tooltip.style.opacity = '1';
            updateTooltipPos(e);
        }

        function updateTooltipPos(e) {
            const tooltip = document.getElementById('chart-tooltip');
            tooltip.style.left = (e.pageX + 15) + 'px';
            tooltip.style.top = (e.pageY - 40) + 'px';
        }

        function hideTooltip() {
            document.getElementById('chart-tooltip').style.opacity = '0';
        }

        function navTo(el, page, index) {
            const marker = document.getElementById('nav-marker');
            const navOverlay = document.getElementById('nav-overlay');
            const view = document.documentElement.getAttribute('data-view');
            
            document.querySelectorAll('.nav-item').forEach(item => item.classList.remove('active'));
            el.classList.add('active');

            const navRect = navOverlay.getBoundingClientRect();
            const elRect = el.getBoundingClientRect();

            if (view === 'mobile') {
                const centerX = elRect.left + elRect.width / 2 - navRect.left;
                const xPercent = (centerX / navRect.width) * 100;
                document.body.style.setProperty('--marker-x', xPercent + '%');
            } else {
                const centerY = elRect.top + elRect.height / 2 - navRect.top;
                document.body.style.setProperty('--marker-y', centerY + 'px');
            }
            
            // No modo Casa (que é este arquivo), o marcador apenas serve de fundo.
        }

        function animateValue(start, end, duration, callback) {
            let startTimestamp = null;
            const step = (timestamp) => {
                if (!startTimestamp) startTimestamp = timestamp;
                const progress = Math.min((timestamp - startTimestamp) / duration, 1);
                const value = progress * (end - start) + start;
                callback(value);
                if (progress < 1) window.requestAnimationFrame(step);
            };
            window.requestAnimationFrame(step);
        }

        function startDueCarousel() {
            const card = document.getElementById('card-vencer');
            const items = document.querySelectorAll('#due-carousel .carousel-item');
            const dotPager = document.getElementById('dot-pager');
            if (!dotPager || items.length === 0) return;
            
            dotPager.innerHTML = '';
            for(let i=0; i<items.length; i++) {
                const dot = document.createElement('div');
                dot.className = 'dot' + (i === 0 ? ' active' : '');
                dotPager.appendChild(dot);
            }

            setInterval(() => {
                const oldItem = items[currentDueIndex];
                if(oldItem){
                    oldItem.classList.add('exit');
                    oldItem.classList.remove('active');
                    setTimeout(() => oldItem.classList.remove('exit'), 1000);
                }
                currentDueIndex = (currentDueIndex + 1) % items.length;
                const newItem = items[currentDueIndex];
                if(newItem) {
                    newItem.classList.add('active');
                    if (newItem.id === 'total-summary-item') card.classList.add('total-active');
                    else card.classList.remove('total-active');
                }
                const dots = dotPager.querySelectorAll('.dot');
                dots.forEach((d, idx) => { d.className = 'dot' + (idx === currentDueIndex ? ' active' : ''); });
            }, 5000); 
        }

        function startHold() {
            const toggle = document.getElementById('main-toggle');
            toggle.classList.add('is-holding');
            const thumb = document.getElementById('thumb');
            const hint = document.getElementById('interaction-hint');
            if(hint) hint.classList.remove('show');
            thumb.style.left = isFaturamentoActive ? '60px' : '4px';
            holdTimer = setTimeout(() => { toggleDashboard(); endHold(); }, holdDuration);
        }

        function endHold() {
            if (typeof holdTimer !== 'undefined') clearTimeout(holdTimer);
            document.getElementById('main-toggle').classList.remove('is-holding');
            document.getElementById('thumb').style.left = ''; 
        }

        function toggleDashboard() {
            isFaturamentoActive = !isFaturamentoActive;
            const widget = document.getElementById('main-toggle');
            widget.classList.toggle('swapped', !isFaturamentoActive);
            const fatSection = document.getElementById('fat-section');
            const credSection = document.getElementById('cred-section');
            if (isFaturamentoActive) {
                fatSection.classList.remove('section-disabled');
                credSection.classList.add('section-disabled');
            } else {
                fatSection.classList.add('section-disabled');
                credSection.classList.remove('section-disabled');
            }
        }

        function navigateShortcut(elementId, targetUrl) {
            const btn = document.getElementById(elementId);
            btn.classList.add('animating');
            setTimeout(() => { 
                btn.classList.remove('animating');
                window.location.href = targetUrl;
            }, 600);
        }

        window.onload = () => {
            const firstItem = document.querySelector('.nav-item');
            navTo(firstItem, 'Casa', 0);
            animateValue(0, currentRevenue, 2000, (v) => {
                const parts = v.toFixed(2).split('.');
                document.getElementById('fat-int').innerText = `R$ ${parseInt(parts[0]).toLocaleString('pt-BR')}`;
                document.getElementById('fat-dec').innerText = `,${parts[1]}`;
            });
            startDueCarousel();
            setTimeout(() => {
                const hint = document.getElementById('interaction-hint');
                if(hint) hint.classList.add('show');
            }, 5000);
        };
    </script>
