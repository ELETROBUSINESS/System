        function updateDashboard() {
            const period = document.getElementById('period-select').value;
            const data = dataSet[period];
            const total = data.current.reduce((a, b) => a + b, 0);
            
            animateValue(0, total, 800, v => document.getElementById('kpi-faturamento').innerText = `R$ ${v.toLocaleString('pt-PT', {maximumFractionDigits: 0})}`);
            animateValue(0, total, 800, v => document.getElementById('chart-total-val').innerText = `R$ ${v.toLocaleString('pt-PT', {minimumFractionDigits: 2})}`);
            animateValue(0, data.ticket, 800, v => document.getElementById('kpi-ticket').innerText = `R$ ${v.toLocaleString('pt-PT', {minimumFractionDigits: 2})}`);
            animateValue(0, data.lucro, 800, v => document.getElementById('kpi-lucro').innerText = `R$ ${v.toLocaleString('pt-PT', {minimumFractionDigits: 0})}`);
            animateValue(0, data.nfce, 800, v => document.getElementById('kpi-nfce-val').innerText = `R$ ${v.toLocaleString('pt-PT', {minimumFractionDigits: 0})}`);
            document.getElementById('kpi-nfce-qtd').innerText = `${data.nfceQty} notas`;

            const catList = document.getElementById('category-list');
            catList.innerHTML = data.cats.map(c => `

            }, 100);

            const labelContainer = document.getElementById('chart-labels');
            labelContainer.innerHTML = data.labels.map(l => `<span>${l}</span>`).join('');
            renderAreaChart(data.current, data.prev);
        }

        function renderAreaChart(values, oldValues) {
            const linePath = document.getElementById('chart-line');
            const fillPath = document.getElementById('chart-fill');
            const dotsGroup = document.getElementById('chart-dots-group');
            const maxVal = Math.max(...values, ...oldValues);
            const stepX = 400 / (values.length - 1);
            dotsGroup.innerHTML = '';
            let d = `M 0 ${200 - (values[0] / maxVal * 160)}`;
            createDot(0, 200 - (values[0] / maxVal * 160), 0, values, oldValues);
            for(let i = 1; i < values.length; i++) {
                const x = i * stepX; const y = 200 - (values[i] / maxVal * 160);
                d += ` L ${x} ${y}`; createDot(x, y, i, values, oldValues);
            }
            linePath.setAttribute('d', d);
            fillPath.setAttribute('d', d + ` L 400 200 L 0 200 Z`);
            linePath.classList.remove('active'); fillPath.classList.remove('active');
            void linePath.offsetWidth; linePath.classList.add('active'); fillPath.classList.add('active');
        }

        function createDot(x, y, index, currentSet, prevSet) {
            const dotsGroup = document.getElementById('chart-dots-group');
            const circle = document.createElementNS("http://www.w3.org/2000/svg", "circle");
            circle.setAttribute("cx", x); circle.setAttribute("cy", y); circle.setAttribute("r", "3.5"); circle.setAttribute("class", "chart-dot");
            circle.addEventListener('mouseenter', (e) => {
                const tooltip = document.getElementById('chart-tooltip');
                const val = currentSet[index]; const oldVal = prevSet[index];
                const diff = ((val - oldVal) / oldVal * 100).toFixed(1);
                tooltip.innerHTML = `R$ ${val.toLocaleString('pt-PT')} <br><span class="${diff >= 0 ? 'text-green-400' : 'text-red-400'}">${diff >= 0 ? '+' : ''}${diff}%</span>`;
                tooltip.style.opacity = '1'; tooltip.style.left = (e.pageX + 10) + 'px'; tooltip.style.top = (e.pageY - 40) + 'px';
            });
            circle.addEventListener('mouseleave', () => document.getElementById('chart-tooltip').style.opacity = '0');
            dotsGroup.appendChild(circle);
        }

        function setViewMode(mode) {
            document.documentElement.setAttribute('data-view', mode);
            setTimeout(() => {
                const activeItem = document.querySelector('.nav-item.active');
                if (activeItem) navTo(activeItem, 'Métricas', 1);
                updateDashboard();
            }, 550);
        }

        function toggleSidebar() {
            const isMin = document.documentElement.getAttribute('data-minimized') === 'true';
            document.documentElement.setAttribute('data-minimized', !isMin);
            document.getElementById('toggle-icon').style.transform = isMin ? 'rotate(0deg)' : 'rotate(180deg)';
            // Reajusta o marcador após a animação de expansão/contração
            setTimeout(() => {
                const activeItem = document.querySelector('.nav-item.active');
                if (activeItem) navTo(activeItem, 'Métricas', 1);
            }, 400);
        }

        function navTo(el, page, index) {
            const marker = document.getElementById('nav-marker');
            const navOverlay = document.getElementById('nav-overlay');
            const view = document.documentElement.getAttribute('data-view');
            const navRect = navOverlay.getBoundingClientRect();
            const elRect = el.getBoundingClientRect();
            if (view === 'mobile') {
                const centerX = elRect.left + elRect.width / 2 - navRect.left;
                document.body.style.setProperty('--marker-x', (centerX / navRect.width * 100) + '%');
            } else {
                const centerY = elRect.top + elRect.height / 2 - navRect.top;
                document.body.style.setProperty('--marker-y', centerY + 'px');
            }
        }

        function animateValue(start, end, duration, callback) {
            let startTimestamp = null;
            const step = (timestamp) => {
                if (!startTimestamp) startTimestamp = timestamp;
                const progress = Math.min((timestamp - startTimestamp) / duration, 1);
                callback(progress * (end - start) + start);
                if (progress < 1) window.requestAnimationFrame(step);
            };
            window.requestAnimationFrame(step);
        }

        window.onload = () => {
            const metricsItem = document.querySelectorAll('.nav-item')[1];
            navTo(metricsItem, 'Métricas', 1);
            updateDashboard();
        };
