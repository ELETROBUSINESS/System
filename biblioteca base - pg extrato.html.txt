    <script>
        // Dados mestre simulados (35 itens para testar paginação)
        const rawData = [];
        for (let i = 1; i <= 35; i++) {
            const isEntry = Math.random() > 0.4;
            rawData.push({
                id: i,
                type: isEntry ? 'in' : 'out',
                name: isEntry ? `Venda #${4800 + i}` : `Pagamento #${1200 + i}`,
                date: `${Math.max(1, 15 - Math.floor(i/2.5))} Jan, 10:30`,
                val: (Math.random() * 800) + 50
            });
        }

        let currentFilter = 'all';
        let currentLimit = 20;

        // Calcula os totais fixos do período uma única vez ou ao carregar dados
        function calculatePeriodTotals() {
            const totalIn = rawData.filter(t => t.type === 'in').reduce((acc, t) => acc + t.val, 0);
            const totalOut = rawData.filter(t => t.type === 'out').reduce((acc, t) => acc + t.val, 0);
            
            document.getElementById('total-in').innerText = `+ R$ ${totalIn.toLocaleString('pt-BR', {minimumFractionDigits: 2})}`;
            document.getElementById('total-out').innerText = `- R$ ${totalOut.toLocaleString('pt-BR', {minimumFractionDigits: 2})}`;
        }

        function renderList() {
            const listEl = document.getElementById('transaction-list');
            const loadMoreBtn = document.getElementById('load-more');
            const searchQuery = document.getElementById('search-input').value.toLowerCase();
            
            // Filtra a lista mas NÃO altera os totais superiores (conforme solicitado)
            let filtered = rawData.filter(t => {
                const matchesType = currentFilter === 'all' || t.type === currentFilter;
                const matchesSearch = t.name.toLowerCase().includes(searchQuery);
                return matchesType && matchesSearch;
            });

            // Gerencia botão "Carregar Mais"
            if (filtered.length > currentLimit) {
                loadMoreBtn.style.display = 'flex';
            } else {
                loadMoreBtn.style.display = 'none';
            }

            const visibleItems = filtered.slice(0, currentLimit);

            if(visibleItems.length === 0) {
                listEl.innerHTML = `<div class="p-10 text-center text-gray-400 text-xs font-bold uppercase tracking-widest">Nenhuma movimentação encontrada</div>`;
                return;
            }

            listEl.innerHTML = visibleItems.map(t => `
                <div class="transaction-item">
                    <div class="trans-icon-box">
                        <svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="3" stroke-linecap="round" stroke-linejoin="round">
                            ${t.type === 'in' ? '<path d="M12 19V5M5 12l7-7 7 7"/>' : '<path d="M12 5v14M5 12l7 7 7-7"/>'}
                        </svg>
                    </div>
                    <div class="flex-1 ml-4 text-left">
                        <p class="text-xs font-bold text-gray-800">${t.name}</p>
                        <p class="text-[9px] text-gray-400 font-bold uppercase tracking-tight">${t.date}</p>
                    </div>
                    <p class="text-xs font-bold ${t.type === 'in' ? 'trans-val-pos' : 'trans-val-neg'}">
                        ${t.type === 'in' ? '+' : '-'} R$ ${t.val.toLocaleString('pt-BR', {minimumFractionDigits: 2})}
                    </p>
                </div>
            `).join('');
        }

        function setFilter(type, el) {
            currentFilter = type;
            currentLimit = 20; 
            document.querySelectorAll('.filter-pill').forEach(p => p.classList.remove('active'));
            el.classList.add('active');
            renderList();
        }

        function filterList() {
            currentLimit = 20; 
            renderList();
        }

        function loadMore() {
            currentLimit += 20;
            renderList();
        }

        function setViewMode(mode) {
            document.documentElement.setAttribute('data-view', mode);
            setTimeout(() => {
                const activeItem = document.querySelector('.nav-item.active');
                if (activeItem) navTo(activeItem, 'Extrato', 3);
            }, 550);
        }

        function toggleSidebar() {
            const isMin = document.documentElement.getAttribute('data-minimized') === 'true';
            document.documentElement.setAttribute('data-minimized', !isMin);
            document.getElementById('toggle-icon').style.transform = isMin ? 'rotate(0deg)' : 'rotate(180deg)';
            setTimeout(() => {
                const activeItem = document.querySelector('.nav-item.active');
                if (activeItem) navTo(activeItem, 'Extrato', 3);
            }, 400);
        }

        function navTo(el, page, index) {
            const navOverlay = document.getElementById('nav-overlay');
            const view = document.documentElement.getAttribute('data-view');
            const navRect = navOverlay.getBoundingClientRect();
            const elRect = el.getBoundingClientRect();
            if (view === 'mobile') {
                const centerX = elRect.left + elRect.width / 2 - navRect.left;
                document.body.style.setProperty('--marker-x', (centerX / navRect.width * 100) + '%');
            } else {
                const centerY = elRect.top + elRect.height / 2 - navRect.top;
                document.body.style.setProperty('--marker-y', centerY + 'px');
            }
        }

        window.onload = () => {
            const extratoItem = document.querySelectorAll('.nav-item')[3];
            navTo(extratoItem, 'Extrato', 3);
            calculatePeriodTotals();
            renderList();
        };
    </script>
